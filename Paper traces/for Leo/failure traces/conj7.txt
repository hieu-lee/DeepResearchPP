\begin{definition}
A positive integer $n$ is \emph{cyclic} if $\gcd\!\bigl(n,\varphi(n)\bigr)=1$, where $\varphi$ is Euler's totient function. Let $C$ be the set of cyclics. Define
\[
L_c:=\Bigl\{k\in\mathbb N:\ \text{for all }j>k,\ \#\bigl\{c\in C:\ j^2<c<(j+1)^2\bigr\}\ge \#\{c\in C:\ k^2<c<(k+1)^2\}\Bigr\},
\]
and let $L_c(n)$ be the $n$th element of $L_c$ in increasing order.
\end{definition}
