\begin{definition}
A positive integer $n$ is \emph{cyclic} if $\gcd\!\bigl(n,\varphi(n)\bigr)=1$, where $\varphi$ is Euler's totient function. Let $(c_n)_{n\ge1}$ be the increasing sequence of cyclics.
\end{definition}
\bigskip
\textbf{Conjecture (Firoozbakht analog for cyclics 3).}\ For every fixed $k\in\mathbb N$ there exists $N(k)$ such that for all $n>N(k)$,
\[
c_n^{\,1/(n+k)}\ >\ c_{n+1}^{\,1/(n+k+1)}.
\]
