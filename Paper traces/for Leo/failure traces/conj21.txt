\begin{definition}
A positive integer $n$ is \emph{cyclic} if $\gcd\!\bigl(n,\varphi(n)\bigr)=1$, where $\varphi$ is Euler's totient function. Let $(\sigma_n)_{n\ge1}$ enumerate the SG cyclics increasingly.
\end{definition}
