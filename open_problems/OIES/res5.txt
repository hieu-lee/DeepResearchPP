\begin{definition}
A positive integer $n$ is called \emph{cyclic} if 
\[
\gcd\bigl(n,\varphi(n)\bigr)=1,
\]
where $\varphi(n)$ is Eulerâ€™s totient function.  Equivalently, every group of order $n$ is cyclic.

Define
\[
C(x)\;:=\;\#\{c\le x:\;c\text{ is cyclic}\}.
\]
Moreover, let
\[
\log_k x
\]
denote the $k$-fold iterated natural logarithm, e.g.\ 
\(\log_2 x = \log(\log x)\), 
\(\log_3 x = \log(\log(\log x))\), etc.
\end{definition}
